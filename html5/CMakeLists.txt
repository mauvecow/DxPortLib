
set(DXPORTLIB_USE_LIBLALIES
    "-s USE_SDL=2 -s USE_SDL_IMAGE=2 -s USE_SDL_TTF=2 -s USE_VORBIS=1 "
)

set(DXPORTLIB_DEBUG_CONFIG
    "-s DISABLE_EXCEPTION_CATCHING=0 -s ASSERTIONS=1 "
)

set(DXPORTLIB_DEPLOY_CONFIG
    "-s FULL_ES3=1 -s SIDE_MODULE=1 -s "
)

set(DXPORTLIB_COMPILE_FLAG
    "-g -O2 -Wall ${ADD_CFLAGS} "
)

string(APPEND DXPORTLIB_COMPILE_FLAG
    ${DXPORTLIB_USE_LIBLALIES}
    ${DXPORTLIB_CXX_CONFIG}
    ${DXPORTLIB_DEBUG_CONFIG}
    ${DXPORTLIB_DEPLOY_CONFIG}
    "-s " SDL2_IMAGE_FORMATS='["png"]'
)

set_target_properties(DxPortLib
    PROPERTIES 
        COMPILE_FLAGS	"${DXPORTLIB_COMPILE_FLAG}"
        LINK_FLAGS		"${DXPORTLIB_COMPILE_FLAG}"
        C_STANDARD      11
        CXX_STANDARD    17
)
